    ```swift
    // SPRECHKRAFT/AppDelegate.swift
    // Zweck: AppKit-Schicht für Menu-Bar-Icon und globalen Hotkey.
    // Implementiert SET-02 (Hotkey ⌥⌘R), SET-05 (Login-Toggle),
    // SET-06 (kein Dock-Icon via .accessory) und FEED-01 (Icon-Zustände via AppState).
    // Quellen: RESEARCH.md Pattern 1, 3, 4 + Code Examples; PATTERNS.md AppDelegate.

    import AppKit
    import SwiftUI
    import KeyboardShortcuts
    import LaunchAtLogin
    import Observation

    /// NotificationCenter.Name für die Brücke AppDelegate → SPRECHKRAFTApp.
    extension Notification.Name {
        static let openSettings = Notification.Name("com.sprechkraft.openSettings")
    }

    @MainActor
    final class AppDelegate: NSObject, NSApplicationDelegate {
        private var statusItem: NSStatusItem!
        var appState: AppState?

        private var observationToken: (any Observable)?

        func applicationDidFinishLaunching(_ notification: Notification) {
            // SET-06: .accessory VOR allem anderen — verhindert Dock-Icon auch
            // wenn LSUIElement aus irgendeinem Grund nicht greift.
            NSApp.setActivationPolicy(.accessory)

            // NSStatusItem-Button mit Split-Click konfigurieren.
            statusItem = NSStatusBar.system.statusItem(withLength: NSStatusItem.squareLength)
            guard let button = statusItem.button else { return }
            button.sendAction(on: [.leftMouseUp, .rightMouseUp])
            button.action = #selector(handleClick(_:))
            button.target = self

            updateIcon()
            observeStateChanges()
            setupHotkey()
        }

        // MARK: - Split-Click Handler

        @objc private func handleClick(_ sender: NSButton) {
            guard let event = NSApp.currentEvent else { return }
            if event.type == .rightMouseUp {
                showMenu()
            } else {
                appState?.toggleRecording()
            }
        }

        // MARK: - NSMenu (temporäres Pattern)

        private func showMenu() {
            let menu = NSMenu()

            // Zeile 1: App-Name, disabled (UI-SPEC D-05)
            let titleItem = NSMenuItem(title: "SPRECHKRAFT", action: nil, keyEquivalent: "")
            titleItem.isEnabled = false
            menu.addItem(titleItem)

            menu.addItem(.separator())

            // Einstellungen…
            let settingsItem = NSMenuItem(
                title: "Einstellungen…",
                action: #selector(openSettingsMenu),
                keyEquivalent: ","
            )
            settingsItem.target = self
            menu.addItem(settingsItem)

            // Beim Login starten — Toggle mit State (SET-05)
            let loginItem = NSMenuItem(
                title: "Beim Login starten",
                action: #selector(toggleLoginItem),
                keyEquivalent: ""
            )
            loginItem.target = self
            loginItem.state = LaunchAtLogin.isEnabled ? .on : .off
            menu.addItem(loginItem)

            menu.addItem(.separator())

            // Beenden
            menu.addItem(NSMenuItem(
                title: "Beenden",
                action: #selector(NSApplication.terminate(_:)),
                keyEquivalent: "q"
            ))

            // KRITISCH (RESEARCH.md Pitfall 1): menu temporär setzen, sofort nach
            // performClick() wieder auf nil — sonst übernimmt AppKit das
            // gesamte Click-Handling und Linksklick würde auch das Menü öffnen.
            statusItem.menu = menu
            statusItem.button?.performClick(nil)
            statusItem.menu = nil
        }

        // MARK: - Menu Actions

        @objc private func openSettingsMenu() {
            // Brücke zur SwiftUI-Scene in SPRECHKRAFTApp.swift.
            // Siehe RESEARCH.md Pitfall 2: openSettings Environment-Action ist
            // auf macOS 26 Tahoe mit .accessory-Policy unzuverlässig.
            NotificationCenter.default.post(name: .openSettings, object: nil)
        }

        @objc private func toggleLoginItem() {
            LaunchAtLogin.isEnabled.toggle()
        }

        // MARK: - Icon-Update via Observation

        /// Beobachtet Änderungen an appState.recordingState und aktualisiert das Icon.
        /// Swift 6 Observation: withObservationTracking feuert bei Zugriff im Closure.
        /// Nach Feuer registrieren wir neu (rekursiv), damit weitere Änderungen auch greifen.
        private func observeStateChanges() {
            withObservationTracking {
                _ = appState?.recordingState
            } onChange: { [weak self] in
                Task { @MainActor [weak self] in
                    self?.updateIcon()
                    self?.observeStateChanges()  // Re-register
                }
            }
        }

        private func updateIcon() {
            let state = appState?.recordingState ?? .idle
            let hostingView = NSHostingView(rootView: StatusBarIconView(state: state))
            hostingView.frame = NSRect(x: 0, y: 0, width: 26, height: 26)

            // Alte Subviews entfernen, neue einsetzen.
            statusItem.button?.subviews.forEach { $0.removeFromSuperview() }
            statusItem.button?.addSubview(hostingView)
            statusItem.button?.frame = hostingView.frame

            // UI-SPEC Accessibility Contract: Label pro Zustand.
            statusItem.button?.setAccessibilityLabel(state.accessibilityLabel)
        }

        // MARK: - Global Hotkey (SET-02)

        private func setupHotkey() {
            KeyboardShortcuts.onKeyUp(for: .toggleRecording) { [weak self] in
                Task { @MainActor [weak self] in
                    self?.appState?.toggleRecording()
                }
            }
        }
    }
    ```

    **Entscheidungsregel für die Observation-Strategie (pflicht, damit SUMMARY eine belastbare Aussage trifft):**

    Teste BEIDE Wege in dieser Reihenfolge:
    1. Baue und starte die App mit der OBEN implementierten Variante (A — withObservationTracking mit rekursiver Re-Registrierung).
    2. Klicke 5× auf das Menu-Bar-Icon. Das Icon MUSS bei jedem Klick die Farbe wechseln.
    3. Falls das Icon nach dem ersten Klick NICHT mehr wechselt (häufig bei gewissen Xcode/Swift-6-Konstellationen, wenn die Re-Registrierung nicht feuert): Entferne `observeStateChanges()` aus `applicationDidFinishLaunching` und rufe statt dessen `updateIcon()` manuell auf:
       - in `handleClick(_:)` NACH `appState?.toggleRecording()`

       Das ist Variante (B) — der robuste Fallback.
    4. Dokumentiere im SUMMARY.md EXPLIZIT:
       - Gewählte Variante: (A) withObservationTracking ODER (B) manueller updateIcon-Aufruf
       - Begründung in einem Satz (z. B. "Variante A funktionierte out-of-the-box mit Xcode 26.0, deshalb gewählt" oder "Variante A aktualisierte nur den ersten Zustandswechsel, deshalb Wechsel zu B")
